      import pandas as pd
      import matplotlib.pyplot as plt
      from statsmodels.tsa.arima.model import ARIMA
      ​
      # Example data creation (you can skip this if you already have your DataFrame)
      data = {
          'Date': pd.date_range(start='2024-01-01', periods=100, freq='D'),
          'precipitation': [1.2, 3.4, 2.5, 0.0, 1.8, 2.2, 0.3, 4.1, 0.0, 1.9] * 10
      }
      df = pd.DataFrame(data)
      df.set_index('Date', inplace=True)
      ​
      # Fitting ARIMA model
      model = ARIMA(df['precipitation'], order=(5, 1, 0))
      arima_result = model.fit()
      ​
      # Getting the forecast for the next 12 steps
      forecast = arima_result.get_forecast(steps=12)
      forecast_index = pd.date_range(start=df.index[-1] + pd.Timedelta(days=1), periods=12, freq='D')
      forecast_mean = forecast.predicted_mean
      forecast_mean = pd.Series(forecast_mean, index=forecast_index)
      ​
      # Getting the confidence intervals
      forecast_ci = forecast.conf_int()
      forecast_ci = pd.DataFrame(forecast_ci, index=forecast_index)
      ​
      # Plotting the historical and forecasted precipitation
      plt.figure(figsize=(14, 7))
      plt.plot(df.index, df['precipitation'], label='Observed')
      plt.plot(forecast_mean.index, forecast_mean, label='Forecast', color='r')
      plt.fill_between(forecast_ci.index, 
                       forecast_ci.iloc[:, 0], 
                       forecast_ci.iloc[:, 1], color='pink', alpha=0.3)
      plt.xlabel('Date')
      plt.ylabel('Precipitation (mm)')
      plt.title('Daily Precipitation Forecast using ARIMA')
      plt.legend()
      plt.show()
